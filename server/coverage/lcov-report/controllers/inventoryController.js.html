// 卸下装备逻辑
export const unequipItem = async (req, res) => {
  try {
    const userId = req.user._id;
    const { slot } = req.body;
    const equipment = await UserEquipment.findOne({ userId });
    if (!equipment || !equipment.slots[slot]) {
      return res.status(400).json({ error: "该槽位没有装备" });
    }
    const itemId = equipment.slots[slot];
 
    // 解除装备状态
    await UserInventory.findByIdAndUpdate(itemId, { equipped: false });
    // 清空槽位
    equipment.slots[slot] = null;
    await equipment.save();
    res.status(200).json({ message: "卸下成功", slot });
  } catch (err) {
    console.error("卸下错误：", err);
    res.status(500).json({ error: "卸下失败" });
  }
};

// 使用消耗品
export const useConsumableItem = async (req, res) => {
  try {
    const userId = req.user._id;
    const inventoryId = req.params.inventoryId;
    
    // 查找背包中的物品
    const inventoryItem = await UserInventory.findById(inventoryId).populate('item');
    if (!inventoryItem) {
      return res.status(404).json({ message: '物品不存在' });
    }
    
    // 检查物品是否属于用户
    if (!inventoryItem.userId.equals(userId)) {
      return res.status(403).json({ message: '无权使用此物品' });
    }
    
    // 检查是否为消耗品
    if (inventoryItem.item.type !== 'consumable') {
      return res.status(400).json({ message: '此物品不是消耗品' });
    }
    
    // 应用消耗品效果
    if (inventoryItem.item.effect === 'heal') {
      const potency = inventoryItem.item.potency || 0;
      
      // 恢复角色状态的HP
      const User = (await import('../models/User.js')).default;
      const user = await User.findById(userId);
      if (user) {
        user.health = Math.min(user.maxHealth || 100, (user.health || 0) + potency);
        await user.save();
      }
      
      // 恢复探索状态的HP
      const dungeonStats = await UserDungeonStats.findOne({ user: userId });
      if (dungeonStats && dungeonStats.currentExploration) {
        const maxHP = dungeonStats.assignedStats.hp || 100;
        dungeonStats.currentExploration.currentHp = Math.min(
          maxHP,
          (dungeonStats.currentExploration.currentHp || 0) + potency
        );
        await dungeonStats.save();
      }
    }
    
    // 减少物品数量
    if (inventoryItem.quantity > 1) {
      inventoryItem.quantity -= 1;
      await inventoryItem.save();
    } else {
      // 如果只有一个，直接从背包中移除
      await inventoryItem.deleteOne();
    }
    
    res.json({ 
      message: `成功使用 ${inventoryItem.item.name}`,
      effect: inventoryItem.item.effect,
      remainingQuantity: inventoryItem.quantity > 1 ? inventoryItem.quantity - 1 : 0
    });
  } catch (error) {
    console.error('使用消耗品失败:', error);
    res.status(500).json({ message: '服务器错误' });
  }
};
